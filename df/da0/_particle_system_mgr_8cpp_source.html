34UParticleObjectComponent* AParticleSystemMgr::GenerateParticle(const FString name, const FVector position, const FRotator rotation,
35 const FVector scale, bool bIsLocalTransform, AActor* parent)
36{
37 auto value = mParticlePool.Find(name);
38 if (value == nullptr)
39 {
40 UE_LOG(LogTemp, Error, TEXT("[AParticleSystemMgr] name invaild."));
41 return nullptr;
42 }
43
44 const auto& particles = (*value)->Particles;
45 for (const auto& particle : particles)
46 {
47 if (particle->IsPlaying())
48 {
49 continue;
50 }
51
52 if(parent != nullptr)
53 {
54 particle->GetOwner()->AttachToComponent(parent->GetRootComponent(), FAttachmentTransformRules::KeepWorldTransform);
55 }
56 particle->StartParticle(position, rotation, scale, bIsLocalTransform);
57 return particle;
58 }
59
60 const auto& newParticle = GenerateParticleBlueprint(*(*value));
61 if(parent != nullptr)
62 {
63 newParticle->GetOwner()->AttachToComponent(parent->GetRootComponent(), FAttachmentTransformRules::KeepWorldTransform);
64 }
65 newParticle->StartParticle(position, rotation, scale, bIsLocalTransform);
66 return newParticle;
67}

69UParticleObjectComponent* AParticleSystemMgr::GenerateParticleBlueprint(UParticlePoolClass& particleSystemStruct) const
70{
71 const auto& bluePrint = particleSystemStruct.Orig;
72 const auto& actor = GetWorld()->SpawnActor<AActor>(bluePrint->GeneratedClass);
73 actor->AttachToComponent(RootComponent, FAttachmentTransformRules::KeepWorldTransform);
74 const auto& particle = Cast<UParticleObjectComponent>(actor->FindComponentByClass<UParticleObjectComponent>());
75 particleSystemStruct.Particles.push_back(particle);
76 return particle;
77}
