15void UParticleObjectComponent::StartParticle(FVector position, FRotator rotation, FVector scale,
16 bool bIsLocalTransform)
17{
18 if (mbIsPlaying)
19 {
20 return;
21 }
22
23 SetActorEnable(true);
24 mbIsPlaying = true;
25 if (bIsLocalTransform)
26 {
27 mActor->SetActorRelativeTransform(FTransform(rotation, position, scale));
28 }
29 else
30 {
31 mActor->SetActorTransform(FTransform(rotation, position, scale));
32 }
33}

35void UParticleObjectComponent::StopParticle()
36{
37 SetActorEnable(false);
38 mbIsPlaying = false;
39 mActor->AttachToComponent(mMgrRootComponent, FAttachmentTransformRules::KeepWorldTransform);
40}

59// Called when the game starts
60void UParticleObjectComponent::BeginPlay()
61{
62 Super::BeginPlay();
63
64 mActor = GetOwner();
65 mParticleSystemComponent = Cast<UParticleSystemComponent>(
66 mActor->GetComponentByClass(UParticleSystemComponent::StaticClass()));
67 mRootComponent = mActor->GetRootComponent();
68 SetActorEnable(false);
69 mbIsPlaying = false;
70}

72// Called every frame
73void UParticleObjectComponent::TickComponent(float DeltaTime, ELevelTick TickType,
74 FActorComponentTickFunction* ThisTickFunction)
75{

77
78 Super::TickComponent(DeltaTime, TickType, ThisTickFunction);
79
80 // 파티클이 끝났으므로 비활성화
81 if (mParticleSystemComponent->HasCompleted())
82 {
83 StopParticle();
84 }
85}
